# Python dependencies for OpenEvolve harness
# Install with: pip install -r evolve/requirements.txt

#   - json: for serialization
#   - random: for evolution operators
#   - logging: for output
#   - pathlib: for file handling
#   - dataclasses: for data structures
#   - re: for parsing benchmark results
#   - argparse: for CLI
